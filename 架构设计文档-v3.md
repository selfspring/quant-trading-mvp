#### 3.2.1 Trace ID 生成规则

```python
import uuid
from datetime import datetime

def generate_trace_id() -> str:
    """
    生成全局唯一的 trace_id
    格式：{timestamp}_{uuid}
    示例：20260309_a1b2c3d4-e5f6-7890-abcd-ef1234567890
    """
    timestamp = datetime.now().strftime('%Y%m%d%H%M%S')
    unique_id = str(uuid.uuid4())
    return f"{timestamp}_{unique_id}"

def generate_child_trace_id(parent_trace_id: str) -> str:
    """
    生成子 trace_id（用于关联上下游事件）
    """
    return f"{parent_trace_id}_child_{uuid.uuid4().hex[:8]}"
```

#### 3.2.2 消息追踪中间件

```python
class MessageTracer:
    """
    消息追踪中间件，自动记录每个进程的事件
    """
    def __init__(self, redis_client, db_client, process_name: str):
        self.redis = redis_client
        self.db = db_client
        self.process_name = process_name
    
    def trace_event(
        self, 
        trace_id: str,
        event_type: str,
        event_data: dict,
        parent_trace_id: str = None,
        status: str = 'success',
        error_msg: str = None
    ):
        """
        记录事件到 message_trace 表
        """
        trace_record = {
            'trace_id': trace_id,
            'timestamp': datetime.now(),
            'process_name': self.process_name,
            'event_type': event_type,
            'event_data': event_data,
            'parent_trace_id': parent_trace_id,
            'status': status,
            'error_msg': error_msg
        }
        
        # 写入数据库
        self.db.insert('message_trace', trace_record)
        
        # 同时写入 Redis（用于实时查询）
        self.redis.setex(
            f"trace:{trace_id}",
            3600,  # 1小时过期
            json.dumps(trace_record)
        )
    
    def publish_with_trace(
        self, 
        channel: str, 
        message: dict,
        parent_trace_id: str = None
    ) -> str:
        """
        发布消息时自动注入 trace_id
        """
        trace_id = generate_trace_id()
        message['trace_id'] = trace_id
        message['parent_trace_id'] = parent_trace_id
        
        # 记录发送事件
        self.trace_event(
            trace_id=trace_id,
            event_type=f'{channel}_published',
            event_data=message,
            parent_trace_id=parent_trace_id
        )
        
        # 发布消息
        self.redis.publish(channel, json.dumps(message))
        
        return trace_id
```

```python
# FastAPI 端点
@app.get("/api/trace/{trace_id}")
def get_trace_chain(trace_id: str):
    """
    查询完整的消息链路
    返回从数据采集到订单成交的所有事件
    """
    # 1. 查询主事件
    main_event = db.query(
        "SELECT * FROM message_trace WHERE trace_id = %s",
        (trace_id,)
    )
    
    # 2. 递归查询所有子事件
    child_events = db.query(
        """
        WITH RECURSIVE trace_chain AS (
            SELECT * FROM message_trace WHERE trace_id = %s
            UNION ALL
            SELECT mt.* FROM message_trace mt
            JOIN trace_chain tc ON mt.parent_trace_id = tc.trace_id
        )
        SELECT * FROM trace_chain ORDER BY timestamp
        """,
        (trace_id,)
    )
    
    return {
        'trace_id': trace_id,
        'events': child_events,
        'total_latency_ms': sum(e['latency_ms'] for e in child_events),
        'status': 'success' if all(e['status'] == 'success' for e in child_events) else 'failed'
    }
```

```python
class CacheInvalidationManager:
    """
    LLM 缓存失效管理器
    """
    # 重大事件关键词（触发缓存清理）
    MAJOR_EVENTS = [
        '美联储', '加息', '降息', '非农', 'CPI', 'GDP',
        '央行', '利率决议', '货币政策', '通胀',
        '黄金储备', '美元指数', '地缘政治'
    ]
    
    def should_invalidate(self, news: NewsItem) -> bool:
        """
        判断是否需要清理缓存
        """
        # 1. 检查是否包含重大事件关键词
        for keyword in self.MAJOR_EVENTS:
            if keyword in news.title or keyword in news.content:
                return True
        
        # 2. 检查新闻重要性
        if news.importance == 'high' and news.confidence > 0.8:
            return True
        
        return False
    
    def invalidate_related_cache(self, news: NewsItem):
        """
        清理相关缓存
        """
        # 1. 提取关键词
        keywords = self.extract_keywords(news.content)
        
        # 2. 在 Chroma 中检索相似新闻
        similar_news = self.chroma.query(
            query_embeddings=[news.embedding],
            n_results=50,  # 清理最相似的 50 条
            where={'importance': {'$in': ['high', 'medium']}}
        )
        
        # 3. 清理 Redis 缓存
        for similar in similar_news['ids'][0]:
```

#### 3.3.2 动态 TTL 策略

```python
class DynamicTTLManager:
    """
    根据市场状态动态调整缓存 TTL
    """
    def get_ttl(self, news: NewsItem, market_regime: str) -> int:
        """
        返回缓存 TTL（秒）
        """
        base_ttl = 6 * 3600  # 基础 6 小时
        
        # 1. 根据新闻重要性调整
        if news.importance == 'high':
            base_ttl = 2 * 3600  # 高重要性：2 小时
        elif news.importance == 'low':
            base_ttl = 24 * 3600  # 低重要性：24 小时
        
        # 2. 根据市场状态调整
        if market_regime == 'volatile':
            base_ttl = base_ttl // 2  # 高波动市：减半
        
        return base_ttl
```

```python
# start_local.py
import subprocess
import sys
import time
import signal

processes = []

def start_process(name, command):
    """启动子进程"""
    print(f"Starting {name}...")
    proc = subprocess.Popen(
        command,
        shell=True,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE
    )
    processes.append((name, proc))
    return proc

def stop_all():
    """停止所有进��"""
    print("\nStopping all processes...")
    for name, proc in processes:
        print(f"Stopping {name}...")
        proc.terminate()
        proc.wait(timeout=5)
    sys.exit(0)

# 注册信号处理
signal.signal(signal.SIGINT, lambda s, f: stop_all())
signal.signal(signal.SIGTERM, lambda s, f: stop_all())

if __name__ == '__main__':
    # 1. 启动 Redis（如果未运行）
    start_process('Redis', 'redis-server')
    time.sleep(2)
    
    # 2. 启动 Chroma（如果未运行）
    start_process('Chroma', 'chroma run --path ./chroma_data')
    time.sleep(2)
    
    # 3. 启动各个进程
    start_process('Data Collector', 'python -m quant.data_collector')
    time.sleep(5)
    
    start_process('Signal Generator', 'python -m quant.signal_generator')
    time.sleep(3)
    
    start_process('Risk Executor', 'python -m quant.risk_executor')
    time.sleep(3)
    
    start_process('Web Server', 'python -m quant.web_server')
    time.sleep(2)
    
    start_process('Monitor', 'python -m quant.monitor')
    
    print("\n✅ All processes started successfully!")
    print("Web interface: http://localhost:8080")
    print("Press Ctrl+C to stop all processes\n")
    
    # 保持主进程运行
```

```python
import requests
import time

def check_process_health():
    """检查所有进程是否正常运行"""
    checks = {
        'Redis': lambda: redis_client.ping(),
        'Chroma': lambda: requests.get('http://localhost:8000/api/v1/heartbeat').status_code == 200,
        'Web Server': lambda: requests.get('http://localhost:8080/health').status_code == 200,

    }
    
    results = {}
    for name, check_fn in checks.items():
        try:
            results[name] = check_fn()
        except Exception as e:
            results[name] = False
            print(f"❌ {name} is down: {e}")
    
    return all(results.values())

# 每 30 秒检查一次
```

## 六、故障处理与降级方案（已完善）

### 6.1 数据源失效

| 场景 | 检测方式 | 降级策略 | 恢复策略 |
|-----|---------|---------|---------|
| CTP 断线 | 心跳超时 | 暂停交易，发送告警 | 自动重连（最多3次） |
| 新闻爬虫失败 | 连续3次失败 | 切换备用源（东方财富 → 新浪） | 每5分钟重试 |
| AKShare 失效 | HTTP 错误 | 切换 TuShare | 每小时重试 |
| LLM API 限流 | 429 错误 | 降级为 neutral，信号权重降低 | 指数退避重试 |

### 6.2 模型失效

| 场景 | 检测方式 | 降级策略 |
|-----|---------|---------|
| ML 模型预测异常 | 置信度 < 0.3 连续10次 | 降低 ML 权重至 20% |
| LLM 解读质量差 | 人工标注准确率 < 60% | 降低 LLM 权重至 10% |
| 技术指标计算错误 | 数值超出合理范围 | 跳过该K线，使用上一根 |

### 6.3 系统异常
